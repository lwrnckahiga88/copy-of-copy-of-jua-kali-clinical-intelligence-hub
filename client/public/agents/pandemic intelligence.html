🔗 Integration Examples

JavaScript Client

class PandemicAPI {
  constructor(apiKey) {
    this.baseURL = 'http://localhost:3000/api/v1';
    this.apiKey = apiKey;
  }

  async getCurrentOutbreak() {
    const response = await fetch(`${this.baseURL}/seird/current`);
    return await response.json();
  }

  async runSimulation(params) {
    const response = await fetch(`${this.baseURL}/seird/simulate`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(params)
    });
    return await response.json();
  }
}

// Usage
const api = new PandemicAPI('your-api-key');
const outbreak = await api.getCurrentOutbreak();

Python Client

import requests

class PandemicAPIClient:
    def __init__(self, api_key):
        self.base_url = "http://localhost:3000/api/v1"
        self.api_key = api_key
        
    def get_risk_assessment(self, region):
        response = requests.get(
            f"{self.base_url}/risk/{region}"
        )
        return response.json()
    
    def simulate_intervention(self, intervention_data):
        response = requests.post(
            f"{self.base_url}/interventions/simulate",
            headers={"Authorization": f"Bearer {self.api_key}"},
            json=intervention_data
        )
        return response.json()

# Usage
client = PandemicAPIClient("your-api-key")
risk_data = client.get_risk_assessment("europe")
