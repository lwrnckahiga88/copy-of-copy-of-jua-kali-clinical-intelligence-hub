<div style="margin-top: 16px; padding: 12px; background: rgba(139, 233, 253, 0.05); border-radius: 6px;">
  <h4 style="margin-top: 0; color: #8be9fd;">Why Are They Mutually Exclusive? The Biological Basis</h4>
  <p>The inverse relationship exists because BRCA1-deficient and HER2-positive cancers develop through fundamentally different mechanisms:</p>
  <ul>
    <li><strong>Different Molecular Pathways:</strong> BRCA1-mutated cancers fall into the Basal-like/TNBC subtype, while HER2-positive cancers form their own distinct subtype with different cells of origin.</li>
    <li><strong>Role of BRCA1 in Differentiation:</strong> BRCA1 protein directs breast cells to mature; its loss locks cells in a primitive, basal-like state (TNBC), not the HER2-positive subtype.</li>
    <li><strong>Alternative Routes to Genomic Instability:</strong> HER2 overexpression causes genomic instability through a different mechanism than BRCA1 loss. One major defect is often enough for carcinogenesis.</li>
  </ul>
</div>

<div style="margin-top: 16px; padding: 12px; background: rgba(80, 250, 123, 0.05); border-radius: 6px;">
  <h4 style="margin-top: 0; color: #50fa7b;">Clinical and Practical Implications</h4>
  <ul>
    <li><strong>Testing is Crucial:</strong> HER2 status is tested on tumor tissue (IHC/FISH), while BRCA1 status is tested via blood/saliva (germline genetic testing).</li>
    <li><strong>Treatment Decisions are Different:</strong> HER2-positive patients receive HER2-targeted therapies; BRCA1-mutated patients may receive PARP inhibitors.</li>
    <li><strong>Family History:</strong> HER2-positive diagnosis doesn't strongly predict BRCA1 mutation, but TNBC diagnosis (especially in younger women) is a strong indicator for BRCA testing.</li>
  </ul>
</div>

  stepCount++; simState.time=Date.now(); simState.exampleGene=Math.random();
  if(simState.tumors.length>0){
    const N=Math.min(simState.tumors.length,10);
    let mat=[]; for(let i=0;i<N;i++){mat[i]=[];for(let j=0;j<N;j++){mat[i][j]=Math.random();}}
    let v=Array(N).fill(1/N);
    for(let k=0;k<10;k++){let w=Array(N).fill(0); for(let i=0;i<N;i++) for(let j=0;j<N;j++) w[i]+=mat[i][j]*v[j]; let norm=Math.sqrt(w.reduce((a,b)=>a+b*b,0)); v=w.map(x=>x/norm);}

    // Adaptive plans
    clusters.forEach(c=>{ if(!c.adaptivePlan) c.adaptivePlan=[]; });
    const allTreatments = ["Drug A","Drug B","Drug C"];
    clusters.forEach(cluster=>{
        const sequences=[];
        function helper(seq){if(seq.length>0) sequences.push([...seq]); if(seq.length>=3) return; allTreatments.forEach(t=>helper([...seq,t]));}
        helper([]);
        let bestSeq=[]; let bestScore=-Infinity;
        sequences.forEach(seq=>{
            let score=0;
            seq.forEach(tName=>{
                const th=cluster.treatmentHistory?.find(h=>h.treatmentName===tName);
                if(th){ const avgScore=th.scores.map(s=>s.reduce((a,b)=>a+b,0)/s.length); score+=avgScore[avgScore.length-1]; }
            });
            if(score>bestScore){ bestScore=score; bestSeq=seq; }
        });
        cluster.adaptivePlan=bestSeq;
    });
    // Update UI
    const list=document.getElementById('adaptivePlanList'); list.innerHTML='';
    clusters.forEach((c,idx)=>{ list.innerHTML+=`<li>Cluster ${idx}: <strong>${c.adaptivePlan.join(" → ")||"No plan yet"}</strong></li>`; });

    // Recommendations
    clusters.forEach(c=>{ if(!c.recommendation) c.recommendation=null; });
    clusters.forEach(cluster=>{
        let bestTreatment=null, bestScore=-Infinity;
        cluster.treatmentHistory?.forEach(th=>{
            const avgScore=th.scores.map(s=>s.reduce((a,b)=>a+b,0)/s.length);
            const improvement=avgScore[avgScore.length-1]-avgScore[0];
            if(improvement>bestScore){ bestScore=improvement; bestTreatment=th.treatmentName; }
        });
        cluster.recommendation=bestTreatment||"No treatment yet";
    });
    const rlist=document.getElementById('recommendationList'); rlist.innerHTML='';
    clusters.forEach((c,idx)=>{ rlist.innerHTML+=`<li>Cluster ${idx}: <strong>${c.recommendation}</strong></li>`; });

    // Render dashboard
    const ctx=document.getElementById('dashboardChart').getContext('2d');
    const datasets=[];
    clusters.forEach((cluster,cIdx)=>{
        cluster.treatmentHistory?.forEach(th=>{
            const avgScores=th.scores.map(s=>s.reduce((a,b)=>a+b,0)/s.length);
            datasets.push({label:`Cluster ${cIdx} - ${th.treatmentName}`,data:avgScores.map((v,i)=>({x:i,y:v})),borderColor:'#'+Math.floor(Math.random()*16777215).toString(16),fill:false,tension:0.3});
        });
    });
    if(window.dashboardChart) window.dashboardChart.destroy();
    window.dashboardChart=new Chart(ctx,{type:'line',data:{datasets},options:{responsive:true,plugins:{legend:{position:'bottom'}},scales:{x:{title:{display:true,text:'Time step'}},y:{min:0,max:1,title:{display:true,text:'Predictive Score'}}}}});
},1000);

</script>
